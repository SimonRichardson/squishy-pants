Router using squishy-pants.js

var search = function() {
        return '/search/:query';
    },
    blog = function() {
        return '/blog/:author/:id';
    };

router()
    .route(
        search,
        function(query) {
            logger('Search Query', query);
        }
    )
    .route(
        blog,
        function(author, id) {
            logger('Blog Author', author);
            logger('Blog Id', id);
        }
    );
